164 void TrajectoryThread::run()
165 {
166  if (waypointsHaveBeenSet && !isPaused) {
167  if (goalAttained() || deactivationLatch)
168  {
169  switch (terminationStrategy)
170  {
171  case BACK_AND_FORTH:
172  flipWaypoints();
173  if (trajType==TIME_OPTIMAL) {
174  setTrajectoryWaypoints(allWaypointList, true);
175  } else {
176  setTrajectoryWaypoints(allWaypoints, true);
177  }
178  break;
179 
180  case REVERSE:
181  if (returningHome) {
182  if (printWaitingNoticeOnce) {
183  std::cout << "Trajectory thread for task: " << task->getTaskName() << " has attained its goal state. Awaiting new commands." << std::endl;
184  printWaitingNoticeOnce = false;
185  }
186  } else {
187  flipWaypoints();
188  if (trajType==TIME_OPTIMAL) {
189  setTrajectoryWaypoints(allWaypointList, true);
190  } else {
191  setTrajectoryWaypoints(allWaypoints, true);
192  }
193  returningHome = true;
194  }
195  break;
196 
197  case REVERSE_STOP:
198  if (returningHome) {
199  stop();
200  } else {
201  flipWaypoints();
202  if (trajType==TIME_OPTIMAL) {
203  setTrajectoryWaypoints(allWaypointList, true);
204  } else {
205  setTrajectoryWaypoints(allWaypoints, true);
206  }
207  returningHome = true;
208  }
209  break;
210 
211  case REVERSE_STOP_DEACTIVATE:
212  if (returningHome) {
213  if(task->deactivate()){
214  isTaskCurrentlyActive = task->isActivated();
215  stop();
216  }else{
217  std::cout << "[WARNING] Trajectory thread for task: " << task->getTaskName() << " has attained its goal state, but cannot be deactivated." << std::endl;
218  yarp::os::Time::delay(1.0); // try again in one second.
219  deactivationDelay += 1.0;
220  if(deactivationDelay >= deactivationTimeout){
221  std::cout << "[WARNING] Deactivation timeout." << std::endl;
222  stop();
223  }
224  }
225  } else {
226  flipWaypoints();
227  if (trajType==TIME_OPTIMAL) {
228  setTrajectoryWaypoints(allWaypointList, true);
229  } else {
230  setTrajectoryWaypoints(allWaypoints, true);
231  }
232  returningHome = true;
233  }
234  break;
235 
236  case CYCLE:
237  cycleWaypoints();
238  if (trajType==TIME_OPTIMAL) {
239  setTrajectoryWaypoints(allWaypointList, true);
240  } else {
241  setTrajectoryWaypoints(allWaypoints, true);
242  }
243  break;
244 
245  case STOP_THREAD:
246  stop();
247  break;
248  case STOP_THREAD_DEACTIVATE:
249  if(task->deactivate()){
250  isTaskCurrentlyActive = task->isActivated();
251  stop();
252  }else{
253  std::cout << "[WARNING] Trajectory thread for task: " << task->getTaskName() << " has attained its goal state, but cannot be deactivated." << std::endl;
254  yarp::os::Time::delay(1.0); // try again in one second.
255  deactivationDelay += 1.0;
256  if(deactivationDelay >= deactivationTimeout){
257  std::cout << "[WARNING] Deactivation timeout." << std::endl;
258  stop();
259  }
260  }
261  break;
262  case WAIT:
263  if (printWaitingNoticeOnce) {
264  std::cout << "Trajectory thread for task: " << task->getTaskName() << " has attained its goal state. Awaiting new commands." << std::endl;
265  printWaitingNoticeOnce = false;
266  }
267  break;
268  case WAIT_DEACTIVATE:
269  if (printWaitingNoticeOnce) {
270  if(task->deactivate()){
271  isTaskCurrentlyActive = task->isActivated();
272 
273  std::cout << "Trajectory thread for task: " << task->getTaskName() << " has attained its goal state. Deactivating task and awaiting new commands." << std::endl;
274  printWaitingNoticeOnce = false;
275  deactivationLatch = true;
276  }else{
277  std::cout << "Trajectory thread for task: " << task->getTaskName() << " has attained its goal state and is awaiting new commands. [WARNING] Could not deactivate the task." << std::endl;
278  yarp::os::Time::delay(1.0); // try again in one second.
279  deactivationDelay += 1.0;
280  if(deactivationDelay >= deactivationTimeout){
281  printWaitingNoticeOnce = false;
282  std::cout << "[WARNING] Deactivation timeout." << std::endl;
283  }
284  }
285  }
286  break;
287  }
288  }
289  else{
290  if (!isTaskCurrentlyActive) {
291  task->activate();
292  isTaskCurrentlyActive = task->isActivated();
293  }
294 
295  desStateBottle.clear();
296 
297  double relativeTime = yarp::os::Time::now() - timeElapsedDuringPause;
298  Eigen::MatrixXd desiredState_tmp;
299 
300  if (trajType==GAUSSIAN_PROCESS)
301  {
302  // trajectory->getDesiredValues(relativeTime, desiredState_tmp, desiredVariance);
303  // desiredState << desiredState_tmp;
304 
305 
306  // for(int i=0; i<desiredState.size(); i++)
307  // {
308  // desStateBottle.addDouble(desiredState(i));
309  // }
310  // #if USING_SMLT
311  // if(useVarianceModulation)
312  // {
313  // Eigen::VectorXd desiredWeights = varianceToWeights(desiredVariance);
314  // for(int i=0; i<desiredWeights.size(); i++)
315  // {
316  // desStateBottle.addDouble(desiredWeights(i));
317  // }
318  // }
319  // #endif
320  }
321  else
322  {
323  // desiredState << trajectory->getDesiredValues(relativeTime);
324  // for(int i=0; i<desiredState.size(); i++)
325  // {
326  // desStateBottle.addDouble(desiredState(i));

369 void TrajectoryThread::flipWaypoints()
370 {
371  if (trajType==TIME_OPTIMAL) {
372  std::list<Eigen::VectorXd> wptListCopy = allWaypointList;
373  allWaypointList.clear();
374  for (auto rit = wptListCopy.rbegin(); rit!= wptListCopy.rend(); ++rit){
375  allWaypointList.push_back(*rit);
376  }
377  goalStateVector = *allWaypointList.rbegin();
378  } else {
379  int nCols = allWaypoints.cols();
380  Eigen::MatrixXd tmp(allWaypoints.rows(), nCols);
381 
382  int c_tmp = nCols - 1;
383  for(int c=0; c<nCols; c++)
384  {
385  tmp.col(c_tmp) = allWaypoints.col(c);
386  c_tmp--;
387  }
388  allWaypoints = tmp;
389  goalStateVector = allWaypoints.rightCols(1);
390  }
391 }

393 void TrajectoryThread::cycleWaypoints()
394 {
395  if (trajType==TIME_OPTIMAL) {
396  allWaypointList.push_front(*allWaypointList.rbegin());
397  allWaypointList.pop_back();
398  goalStateVector = *allWaypointList.rbegin();
399  } else {
400  int nCols = allWaypoints.cols();
401  Eigen::MatrixXd tmp(allWaypoints.rows(), nCols);
402 
403  tmp.col(0) = allWaypoints.rightCols(1);
404  tmp.rightCols(nCols-1) = allWaypoints.leftCols(nCols-1);
405  allWaypoints = tmp;
406  goalStateVector = allWaypoints.rightCols(1);
407  }

410 bool TrajectoryThread::setTrajectoryWaypoints(const Eigen::MatrixXd& userWaypoints, bool containsStartingWaypoint)
411 {
412  Eigen::MatrixXd _userWaypoints = userWaypoints; // Copy waypoints first in case we use allWaypoints as an arg.
413  if (weightDimension==_userWaypoints.rows())
414  {
415  if(containsStartingWaypoint)
416  {
417  allWaypoints = _userWaypoints;
418  }
419  else // Add starting waypoint
420  {
421  allWaypoints = Eigen::MatrixXd(weightDimension, _userWaypoints.cols()+1);
422  allWaypoints.col(0) << getCurrentTaskStateAsVector();
423  for(int i=0; i<_userWaypoints.cols(); i++)
424  {
425  allWaypoints.col(i+1) << _userWaypoints.col(i);
426  }
427  }
428 
429  goalStateVector = allWaypoints.rightCols(1);
430 
431  trajectory->setWaypoints(allWaypoints);
432 
433  #if USING_SMLT
434  if (trajType==GAUSSIAN_PROCESS)
435  {
436  maximumVariance = std::dynamic_pointer_cast<ocra::GaussianProcessTrajectory>(trajectory)->getMaxVariance();
437  }
438  #endif
439 
440  printWaitingNoticeOnce=true;
441  deactivationLatch = false;
442  waypointsHaveBeenSet = true;
443  return true;
444  }
445  else
446  {
447  std::cout << "[ERROR](TrajectoryThread::setTrajectoryWaypoints): The dimension (# DOF) of the waypoints you provided, " << _userWaypoints.rows() << ", does not match the dimension of the task, " << weightDimension <<". Thread not starting." << std::endl;
448  return false;
449  }
