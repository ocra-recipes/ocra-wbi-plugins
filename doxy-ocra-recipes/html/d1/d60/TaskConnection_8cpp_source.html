365 bool TaskConnection::openControlPorts()
366 {
367  bool portsConnected = true;
368 
369  yarp::os::Bottle message, reply;
370  message.addInt(ocra::TASK_MESSAGE::OPEN_CONTROL_PORTS);
371  this->taskRpcClient.write(message, reply);
372  if(reply.get(0).asInt() == ocra::TASK_MESSAGE::OCRA_SUCCESS) {
373  message.clear();
374  reply.clear();
375  message.addInt(ocra::TASK_MESSAGE::GET_CONTROL_PORT_NAMES);
376  this->taskRpcClient.write(message, reply);
377  this->taskInputPortName = reply.get(0).asString();
378  this->taskOutputPortName = reply.get(1).asString();
379 
380  this->inputPortName = "/TaskConnection/"+std::to_string(taskConnectionNumber)+"/"+this->taskName+":i";
381  this->outputPortName = "/TaskConnection/"+std::to_string(taskConnectionNumber)+"/"+this->taskName+":o";
382 
383  portsConnected = portsConnected && inputPort.open(this->inputPortName.c_str());
384  portsConnected = portsConnected && outputPort.open(this->outputPortName.c_str());
385 
386  this->inpCallback = std::make_shared<inputCallback>(*this);
387  inputPort.setReader(*(this->inpCallback));
388 
389 
390  if (portsConnected) {
391  portsConnected = portsConnected && yarp.connect(this->taskOutputPortName.c_str(), this->inputPortName.c_str());
392  portsConnected = portsConnected && yarp.connect(this->outputPortName.c_str(), this->taskInputPortName.c_str());
393 
394  this->controlPortsAreOpen = portsConnected;
395  if (!this->controlPortsAreOpen) {
396  yLog.error() << "Could not open the control ports!";
397  }
398  } else {
399  return false;
400  }
401  } else {
402  return false;
403  }
404 
405  return portsConnected;
406 }
