416  void Task::setAutoGains(double freq, double massSaturation)
417  {
418  JacobiSVD<MatrixXd>& svd = pimpl->gainsWorkspace.svd;
419  svd.compute(getDesiredMass(), ComputeFullU | ComputeFullV);
420 
421  const VectorXd& s = svd.singularValues();
422  pimpl->gainsWorkspace.s = (s.array() < massSaturation).select(s, massSaturation);
423 
424  const double w = 2. * M_PI * freq;
425  VectorXd& kps = pimpl->gainsWorkspace.kps;
426  VectorXd& kds = pimpl->gainsWorkspace.kds;
427 
428  kps = s * w*w; // kp[i] = s[i] * w^2
429  kds = (kps.array()*s.array()).sqrt() * 2.; // kd[i] = 2 * sqrt(kp[i]*s[i])
430 
431  pimpl->K = svd.matrixU() * kps.asDiagonal() * svd.matrixV().transpose();
432  pimpl->B = svd.matrixU() * kds.asDiagonal() * svd.matrixV().transpose();
433  }
